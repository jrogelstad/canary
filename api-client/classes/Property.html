Module: Property

A get/set function. Includes state handling for events.

Example:

let p = f.prop();

p() //
p("foo") // foo
p() // foo

// Disable the property
p.state().send("disable");
p("bar") // foo
p() // foo

// Enable the property
p.state().send("enable");
p("bar") // bar
p() // bar

// Respond to change events
p.state().resolve("/Changing").enter(function () {
    let msg = (
        "Value was \"" + p.oldValue() + "\", " +
        "will be \"" + p.newValue() + ".\""
    );
    console.log(msg);
})
p("Hello World"); // Value was "bar", will be "Hello World."
                  // Hello World

// Silence change events
p.send("silence");
p("Foo bar)" // Foo bar

// Enable change events again
p.send("report");
p("Moo") // Value was "Foo bar", will be "Moo."
         // Moo
